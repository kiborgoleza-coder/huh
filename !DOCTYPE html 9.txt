function toggleAnswer(index) {
    const ans = document.getElementById('q-ans-' + index);
    const card = ans.closest('.question-card');
    if (ans.classList.contains('show')) {
        ans.classList.remove('show');
        card.classList.remove('revealed');
    } else {
        ans.classList.add('show');
        card.classList.add('revealed');
    }
}

// ===== QUIZ =====
function renderQuiz() {
    if (currentQuiz >= quizQuestions.length) {
        showQuizFinish();
        return;
    }
    
    const q = quizQuestions[currentQuiz];
    document.getElementById('quiz-current').textContent = currentQuiz + 1;
    document.getElementById('quiz-total').textContent = quizQuestions.length;
    document.getElementById('quiz-progress-fill').style.width = ((currentQuiz / quizQuestions.length) * 100) + '%';
    
    document.getElementById('quiz-target').innerHTML = `
        ${q.target}
        <span class="pinyin-hint">${q.pinyin} — что это?</span>
    `;
    
    const optsDiv = document.getElementById('quiz-options');
    optsDiv.innerHTML = q.options.map(opt => `
        <button class="quiz-option" onclick="checkAnswer('${opt}', '${q.correct}')">${opt}</button>
    `).join('');
    
    document.getElementById('quiz-feedback').className = 'quiz-feedback';
    document.getElementById('quiz-feedback').textContent = '';
    document.getElementById('next-btn').classList.remove('show');
}

function checkAnswer(selected, correct) {
    const options = document.querySelectorAll('.quiz-option');
    const feedback = document.getElementById('quiz-feedback');
    
    options.forEach(btn => {
        btn.classList.add('disabled');
        if (btn.textContent === correct) btn.classList.add('correct');
        else if (btn.textContent === selected && selected !== correct) btn.classList.add('wrong');
    });
    
    if (selected === correct) {
        quizScore++;
        feedback.className = 'quiz-feedback show correct-fb';
        feedback.innerHTML = '✅ Правильно! Молодец!';
        createConfetti();
    } else {
        feedback.className = 'quiz-feedback show wrong-fb';
        feedback.innerHTML = `❌ Неверно. Правильно: <b>${correct}</b>`;
    }
    
    document.getElementById('next-btn').classList.add('show');
}

function nextQuestion() {
    currentQuiz++;
    renderQuiz();
}

function showQuizFinish() {
    const card = document.getElementById('quiz-card');
    const percentage = Math.round((quizScore / quizQuestions.length) * 100);
    
    card.innerHTML = `
        <div class="finish-screen">
            <div class="trophy">🏆</div>
            <h2>Викторина завершена!</h2>
            <div class="score-display">${quizScore} / ${quizQuestions.length}</div>
            <p style="color: var(--gray); font-size: 1.1rem;">
                ${percentage >= 80 ? 'Отлично! Ты мастер! 🎉' : percentage >= 50 ? 'Неплохо, продолжай учить! 💪' : 'Попробуй ещё раз! 📚'}
            </p>
            <button class="restart-btn" onclick="restartQuiz()">Пройти снова ↺</button>
        </div>
    `;
    if (percentage >= 60) createConfetti();
}

    renderQuiz();
}

// ===== MATCH GAME =====
function renderMatch() {
    const grid = document.getElementById('match-grid');
    const selected = vocabulary.slice(0, 8);
    const cards = [];
    
    selected.forEach(v => {
        cards.push({ type: 'char', value: v.char, pair: v.char });
        cards.push({ type: 'meaning', value: v.meaning, pair: v.char });
    });
    
    cards.sort(() => Math.random() - 0.5);
    
    grid.innerHTML = cards.map((c, i) => `
        <div class="match-card" data-pair="${c.pair}" data-index="${i}" onclick="handleMatchClick(this)">
            <div class="match-char">${c.type === 'char' ? c.value : '?'}</div>
            <div class="match-sub">${c.type === 'meaning' ? c.value : c.value}</div>
        </div>
    `).join('');
}

function handleMatchClick(card) {
    if (matchLocked || card.classList.contains('matched')) return;
    
    card.classList.add('selected');
    
    if (!matchSelected) {
        matchSelected = card;
        return;
    }
    
    if (matchSelected === card) {
        card.classList.remove('selected');
        matchSelected = null;
        return;
    }
    
    matchLocked = true;
    matchMoves++;
    document.getElementById('match-moves').textContent = matchMoves;
    
    const pair1 = matchSelected.dataset.pair;
    const pair2 = card.dataset.pair;
    
    if (pair1 === pair2) {
        matchSelected.classList.remove('selected');
        matchSelected.classList.add('matched');
        card.classList.remove('selected');
        card.classList.add('matched');
        matchPairsFound++;
        document.getElementById('match-score').textContent = matchPairsFound;
        matchSelected = null;
        matchLocked = false;
        
        if (matchPairsFound === 8) {
            setTimeout(() => {
                alert('🎉 Все пары найдены за ' + matchMoves + ' ходов!');
                createConfetti();
            }, 500);
        }
    } else {
        setTimeout(() => {
            matchSelected.classList.remove('selected');
            card.classList.remove('selected');
            matchSelected = null;
            matchLocked = false;
        }, 800);
    }
}

// ===== FILL BLANKS =====
function renderFill() {
    const list = document.getElementById('fill-list');
    list.innerHTML = fillBlanks.map((f, i) => `
        <div class="fill-card">
            <div class="fill-sentence" id="fill-sent-${i}">
                ${f.sentence.replace('___', '<span class="blank" id="blank-' + i + '">?</span>')}
            </div>
            <div style="color: var(--gray); margin-bottom: 1rem;">${f.trans}</div>
            <div class="fill-options" id="fill-opts-${i}">
                ${f.options.map(opt => `
                    <button class="fill-option" onclick="checkFill(${i}, '${opt}', '${f.blank}')">${opt}</button>
                `).join('')}
            </div>
            <div class="quiz-feedback" id="fill-fb-${i}" style="margin-top: 1rem;"></div>
        </div>
    `).join('');
}

function checkFill(index, selected, correct) {
    const blank = document.getElementById('blank-' + index);
    const fb = document.getElementById('fill-fb-' + index);
    const opts = document.querySelectorAll('#fill-opts-' + index + ' .fill-option');
    
    opts.forEach(btn => btn.classList.add('used'));
    
    blank.textContent = selected;
    
    if (selected === correct) {
        blank.style.color = 'var(--success)';
        blank.style.borderColor = 'var(--success)';
        fb.className = 'quiz-feedback show correct-fb';
        fb.innerHTML = '✅ Правильно!';
        createConfetti();
    } else {
        blank.style.color = 'var(--danger)';
        blank.style.borderColor = 'var(--danger)';
        fb.className = 'quiz-feedback show wrong-fb';
        fb.innerHTML = `❌ Неверно. Правильно: <b>${correct}</b>`;
        
        setTimeout(() => {
            blank.textContent = correct;
            blank.style.color = 'var(--success)';
            blank.style.borderColor = 'var(--success)';
        }, 1000);
    }
}

// ===== CONFETTI =====
function createConfetti() {
    const colors = ['#7c3aed', '#2563eb', '#f59e0b', '#10b981', '#ef4444', '#ec4899'];
    for (let i = 0; i < 40; i++) {
        const conf = document.createElement('div');
        conf.className = 'confetti';
        conf.style.left = Math.random() * 100 + 'vw';
        conf.style.background = colors[Math.floor(Math.random() * colors.length)];
        conf.style.animationDuration = (Math.random() * 2 + 2) + 's';
        conf.style.borderRadius = Math.random() > 0.5 ? '50%' : '0';
        document.body.appendChild(conf);
        setTimeout(() => conf.remove(), 4000);
    }
}
